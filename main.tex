  function calcModifier(score) {
    if (score == "") { return ""; }
    else {
      var mod = Math.floor((score-10)/2);
      if (mod >= 0) { return "+" + mod; }
      else if (mod < 0) { return mod; }
      else { return ""; }
    }
  }
  function calcProfBonus() {
    var level = this.getField("lvl").value;
    if (level == "") { 
      this.getField("prof-bonus").value = "";
    }
    else {
      var bonus = 2 + Math.floor((level-1)/4);
      this.getField("prof-bonus").value = "+" + bonus;
    }
  }
  function calcSkillMod(ability_mod, add_proficiency) {
    if (ability_mod == "") { return ""; }
    else {
      var mod = ability_mod;
      if (add_proficiency) {
        var profbonus = 0+this.getField("prof-bonus").value;
        mod += profbonus;
      } 
      return mod;
    }
  }

  function formatBonus() {
    var mod = event.value;
    if (mod == "+" || mod == "-" || mod == "") { event.value = "" }
    else if (mod >= 0) { event.value = "+" + mod; }
    else if (mod < 0) { event.value = mod; }
    else { event.value = ""; }
  }
  function formatLevel() {
    var level = event.value;
    if (level == "") { event.value = ""; return; }
    if (level > 20) { level = 20; }
    if (level < 1) {level = 1; }
    event.value = level;
  }

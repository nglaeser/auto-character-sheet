  function calcModifier() {
    if (this.getField("str-score").value == "") { event.value = "" }
    else {
      var score = 0+this.getField("str-score").value;
      var mod = Math.floor((score-10)/2);
      if (mod >= 0) { event.value = "+" + mod; }
      else if (mod < 0) { event.value = mod; }
      else { event.value = ""; }
    }
  }
